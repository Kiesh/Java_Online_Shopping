CREATE TABLE ECO_USER_CREDENTIALS
(

USERID VARCHAR2(6) PRIMARY KEY,
PASSWORD VARCHAR2(20) NOT NULL,
USERTYPE VARCHAR2(1) CHECK (USERTYPE='A' OR USERTYPE='S' OR USERTYPE='C'),
LOGINSTATUS NUMBER(1) CHECK (LOGINSTATUS=1 OR LOGINSTATUS=0)

);

CREATE sequence  ECO_SEQ_USERID
START WITH 1000
INCREMENT BY 1;



create table ECO_USER_REGISTER
(
Userid VARCHAR2(6), Foreign Key (Userid) references ECO_USER_CREDENTIALS(Userid),
Firstname VARCHAR2(20) Not Null,
Lastname VARCHAR2(20) Not Null,
DateOfBirth DATE Not Null,
Gender VARCHAR2(7) Not Null,
Street VARCHAR(30) Not Null,
Location VARCHAR2(20) Not Null,
City VARCHAR2(20) Not Null,
State VARCHAR2(20) Not Null,
Zip VARCHAR2(10),
MobileNo VARCHAR(10) check (length(MobileNo) = 10),
EmailId VARCHAR2(30) Not Null
);
CREATE TABLE ECO_USER_CREDIANTLS_PRODUCT
(
ID VARCHAR2(25) PRIMARY KEY,
NAME VARCHAR2(25) NOT NULL,
PRICE NUMBER(7,2),
IMAGE BLOB
);
 CREATE TABLE ECO_USER_CREDIANTLS_CART
(
ID VARCHAR2(200) NOT NULL,
NAME VARCHAR2(200) NOT NULL,
QTY NUMBER(2,4) NOT NULL,
UNIT_PRICE(2,4) NOT NULL

);
